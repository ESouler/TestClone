# !/bin/bash
echo "please enter two numbers"
read a
read c

echo "Please choose an operation"
echo "Add"
echo "Subtract"
echo "Divide"
echo "Multiply"
read b

if [ $b = "1" ]
   then
      echo $((a+b))
elif [ $b = "2" ]
   then
      echo $((a-b))
elif [ $b = "3" ]
   then
      echo $((a/b))
elif [ $b = "4" ]
   then
       echo $((a*b))
fi
exit
